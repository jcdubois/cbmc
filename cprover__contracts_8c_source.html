783#if __CPROVER_DFCC_DEBUG_LIB

785{
786__CPROVER_HIDE:;

808# pragma CPROVER check disable "pointer"

811# pragma CPROVER check disable "signed-overflow"
812# pragma CPROVER check disable "undefined-shift"
813# pragma CPROVER check disable "conversion"

816 "ptr NULL or writable up to size");

849 (int)elem->is_writable & (int)__CPROVER_same_object(elem->lb, ptr) &
850 (__CPROVER_POINTER_OFFSET(elem->lb) <= offset) &
851 (__CPROVER_POINTER_OFFSET(ub) <= __CPROVER_POINTER_OFFSET(elem->ub));
852 ++elem;
853 --idx;
854 }
855 return incl;
856# pragma CPROVER check pop
